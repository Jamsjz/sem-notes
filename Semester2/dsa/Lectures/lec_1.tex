\section{Introduction to Algorithm}
\label{sec:intro-algo}

\begin{definition}[Algorithm]
  An algorithm is a step-by-step procedure or set of rules desighned to solve a
  problem or perform a task.
\end{definition}

\textbf{Algorithms are everywhere}
\begin{itemize}
  \item Sorting a list of names alphabetically.
  \item Searching for a word in a dictionary.
\end{itemize}

\section{Characterstics of an Algorithm}
\label{ssec:char-algo}
\paragraph{Finiteness} The algorithm should terminate.
\paragraph{Definiteness} Every step should be well-defined (non-ambiguos)
\paragraph{Input and Input Size}
The no. of entities the algorithm handles. It is denoted by "n".
\begin{eg}
  In a searching algorithm,the algorithm needs a sequence of entities and the
  entity to search.
\end{eg}
\paragraph{Output} The outcome of running the whole algorithm.

\paragraph{Effectiveness (not to be confused with efficient)}
Does the algorithm solve the problem it was meant to
solve? \autoref{ssec:components-of-algo}

\paragraph{Efficiency}

real world scenarios.

\section{Analysis of an Algorithm}
\label{ssec:analysis-algo}
\begin{problem}[\textbf{Why to Analyze Algorithm?}]
  Some sorting algoritms are:
  \begin{itemize}
    \item Quick
    \item Merge
    \item Bubble
    \item Selection
  \end{itemize}
  \textbf{How to know which sorting algorithm is fastest?}\\ 
  \textbf{How to know which sorting algorithm takes the least amount of memory?\\}
  \begin{answer}
    Analyze the algorithm!! for \textbf{Efficiency}\\
    Basis of measuring efficiency:
    \begin{enumerate}
      \item Time Complexity
      \item Space Complexity
    \end{enumerate}
  \end{answer}
\end{problem}
