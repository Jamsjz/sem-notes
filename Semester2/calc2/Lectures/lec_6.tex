\chapter{Relation Between Cartesian, Polar, Cylindrical and Spherical
Coordinates}
\label{chap:relation-between-cartesian-polar-cylindrical-and-spherical-coordinates}
\lecture{6}{10 May. 22:42}{Sixth Lecture}

\begin{equation}
  \label{eqn:cart-cylind-sphere-relation}
  \begin{displaystyle}
    \rho^2 = r^2+z^2 = x^2+y^2+z^2
  \end{displaystyle}
\end{equation}

\begin{equation}
  \label{eqn:cylin-sphere-relation}
  \begin{aligned}
    z &= \rho \cos \phi \\
    r &= \rho \sin \phi
  \end{aligned}
\end{equation}
