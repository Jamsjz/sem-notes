\chapter{Ctrical Points in Two Variable Functions}
\label{chap:notable-points-in-two-variable-functions}
\lecture{11}{4 Jun. 11:39}{Eleventh Lecture}

A function \(z=f(x,y)\) is a function with two variables. It may have
\emph{Local or Global \textbf{Minima} or \textbf{Maxima} or
\textbf{Saddle Point}}.
One can also check the concavity of the surface.

\section{Critical Points}
\label{sec:critical-points}
A critical point is a point where \(f_x=f_y=0\) or \(f_x\) or \(f_y\) is
undefined.

\section{Saddle Points}
\label{sec:saddle-points}
If a point \((a,b,f(a,b))\) is a \href{sec:critical-points}{critical point} and
if \[ \forall (x-a)^2 + (y-b)^2 = r^2, \]
\[\exists (x,y) \text{ s.t. } f(x,y)>f(a,b) \text{ somewhere } \land
f(x,y)<f(a,b) \text{ elsewhere }\]
Then, \((a,b,f(a,b))\) is a saddle point.

\section{Hessin/Discriminant of \(f\): \(D_f\)}
\label{sec:hessin-discriminant-of-f-d_f}
\[ 
  D_f = 
  \begin{bmatrix} 
    f_{xx}&f_{xy}\\
    f_{yx}&f_{yy}
  \end{bmatrix} 
\] 

\section{Confirming Extremes or Saddle}
\label{sec:confirming-extremes-or-saddle}

\begin{align}
  \textbf{Local Maxima }&: f_{xx}<0 \text{ and } D_f
  >0)\label{eqn:local-maxima-2vf}\\
  \textbf{Local Minima }&: f_{xx}>0 \text{ and } D_f<0)
  \label{eqn:local-minima-2vf}\\
  \textbf{Saddle }&: f_{xx}=0 \text{ and } D_f<0) \label{eqn:saddle-2vf}\\
  \textbf{Inconclusive }&:  D_f=0) \label{eqn:inconclusive-2vf}
\end{align}
